Ecma Script : 자바스크립트의 버전을 나타내는 표준 정의어 es2, es4, /es6 표준/, es8버전
	
	script에서 html요소로 접근하기
	
	id를 이용
	 - document.getElementById('id') //es2버전, 더 효율적이다.
	 - document.querySelector('#id') //es6버전, new 버전
	 
	tag명 활용
	 - document.getElementsByTagName('태그명'[n])		//n번째 태그 선택
	 - document.getElementsByTagNmae('태그명')	//모든 '태그명' 선택
	 - document.querySelector('태그명')		//첫번째 '태그명'만 선택
	 - document.querySelectorAll('태그명')		//모든 '태그명' 선택
	 
	class명 활용
	 - document.getElementByClassName('클래스명')[n]	//n번째 클래스 선택
	 - ducumnet.getElementByClassName('클래스명')		//모든 '클래스명' 선택
	 - document.querySelector('.클래스명')			//첫번째 '클래스명'만 선택
	 - document.querySelectorAll('.클래스명')		//모든 '클래스명' 선택
